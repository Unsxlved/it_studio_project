                double totalMiles = 0;
                double weeklyAverageMiles;

                }

                weeklyAverageMiles = totalMiles / NUM_DAYS;

                System.out.println("You have travelled a total of " + totalMiles +" miles this week");
                System.out.printf("\nYour daily average for this week is: %.2f " ,  weeklyAverageMiles);
